    # 一、客户端Client编程实现

    # 1.1创建功能包

    在src目录下使用命令创建功能包

    catkin_create_pkg learning_service roscpp rospy std_msgs geometry_msgs turtlesim

    # 1.2编写功能

    完成功能包创建后编写功能

    /**
 * 该例程将请求/spawn服务，服务数据类型turtlesim::Spawn
 */
#include <ros/ros.h>
#include <turtlesim/Spawn.h>

int main(int argc, char** argv)
{
    // 初始化ROS节点
	ros::init(argc, argv, "turtle_spawn");
    // 创建节点句柄
	ros::NodeHandle node;
    // 发现/spawn服务后，创建一个服务客户端，连接名为/spawn的service
	ros::service::waitForService("/spawn");
	ros::ServiceClient add_turtle = node.serviceClient<turtlesim::Spawn>("/spawn");
    // 初始化turtlesim::Spawn的请求数据
	turtlesim::Spawn srv;
	srv.request.x = 2.0;
	srv.request.y = 2.0;
	srv.request.name = "turtle2";
    // 请求服务调用
	ROS_INFO("Call service to spwan turtle[x:%0.6f, y:%0.6f, name:%s]", 
			 srv.request.x, srv.request.y, srv.request.name.c_str());

	add_turtle.call(srv);
	// 显示服务调用结果
	ROS_INFO("Spwan turtle successfully [name:%s]", srv.response.name.c_str());
	return 0;
};

    # 1.2.1代码解读

    • 初始化节点
    // 初始化ROS节点
ros::init(argc, argv, "turtle_spawn");
    • 创建句柄管理资源
     // 创建节点句柄
ros::NodeHandle node;
    • 查询当前服务中是否存在服务，如果没有的话就会一直等待
    ros::service::waitForService("/spawn");

    这是一种阻塞型的API，循环等待spawn服务

    • 创建客户端add_turtle
    ros::ServiceClient add_turtle = node.serviceClient<turtlesim::Spawn>("/spawn");

    该客户端向spawn客户端发送请求("/spawn")，请求的数据类型<turtlesim::Spawn>

    • 请求数据
    // 初始化turtlesim::Spawn的请求数据
turtlesim::Spawn srv;
srv.request.x = 2.0;
srv.request.y = 2.0;
srv.request.name = "turtle2";

    turtlesim::Spawn srv;作用是定义一个请求数据，后续对数据进行封装

    • 发送log数据
    // 请求服务调用
ROS_INFO("Call service to spwan turtle[x:%0.6f, y:%0.6f, name:%s]", 
		 srv.request.x, srv.request.y, srv.request.name.c_str());
    • 发送请求数据
    add_turtle.call(srv);

    注意：call()方法也是一个阻塞性的函数，执行该函数后会不断等待客户端给予响应

    • 反馈结果
    // 显示服务调用结果
ROS_INFO("Spwan turtle successfully [name:%s]", srv.response.name.c_str());

    • 初始化ROS节点
    • 创建一个Client实例
    • 发布服务请求数据
    • 等待Server处理之后的应答结果

    # 二、服务端Server编程实现

    # 2.1编写功能

    /**
 * 该例程将执行/turtle_command服务，服务数据类型std_srvs/Trigger
 */
#include <ros/ros.h>
#include <geometry_msgs/Twist.h>
#include <std_srvs/Trigger.h>

ros::Publisher turtle_vel_pub;
bool pubCommand = false;

// service回调函数，输入参数req，输出参数res
bool commandCallback(std_srvs::Trigger::Request  &req,
         			std_srvs::Trigger::Response &res)
{
	pubCommand = !pubCommand;
    // 显示请求数据
    ROS_INFO("Publish turtle velocity command [%s]", pubCommand==true?"Yes":"No");
	// 设置反馈数据
	res.success = true;
	res.message = "Change turtle command state!"
    return true;
}

int main(int argc, char **argv)
{
    // ROS节点初始化
    ros::init(argc, argv, "turtle_command_server");
    // 创建节点句柄
    ros::NodeHandle n;
    // 创建一个名为/turtle_command的server，注册回调函数commandCallback
    ros::ServiceServer command_service = n.advertiseService("/turtle_command", commandCallback);
	// 创建一个Publisher，发布名为/turtle1/cmd_vel的topic，消息类型为geometry_msgs::Twist，队列长度10
	turtle_vel_pub = n.advertise<geometry_msgs::Twist>("/turtle1/cmd_vel", 10);
    // 循环等待回调函数
    ROS_INFO("Ready to receive turtle command.");
	// 设置循环的频率
	ros::Rate loop_rate(10);
	while(ros::ok())
	{
		// 查看一次回调函数队列
    	ros::spinOnce();
		// 如果标志为true，则发布速度指令
		if(pubCommand)
		{
			geometry_msgs::Twist vel_msg;
			vel_msg.linear.x = 0.5;
			vel_msg.angular.z = 0.2;
			turtle_vel_pub.publish(vel_msg);
		}
		//按照循环频率延时
	    loop_rate.sleep();
	}
    return 0;
}

    # 2.1.1代码解读

    • 初始化节点
    // ROS节点初始化
ros::init(argc, argv, "turtle_command_server");
    • 创建句柄管理资源
    // 创建节点句柄
ros::NodeHandle n;
    • 创建service
    // 创建一个名为/turtle_command的server，注册回调函数commandCallback
ros::ServiceServer command_service = n.advertiseService("/turtle_command", commandCallback);

    "/turtle_command"是自定义的一个service名

    通过回调函数来处理收到Client的信息，这一点和Subscriber机制非常相似

    • 创建Publisher，在这里Publisher的主要作用就是给海龟发送速度指令，让海龟动起来
    // 创建一个Publisher，发布名为/turtle1/cmd_vel的topic，消息类型为geometry_msgs::Twist，队列长度10
turtle_vel_pub = n.advertise<geometry_msgs::Twist>("/turtle1/cmd_vel", 10);
    • 展示日志信息和设置循环频率
    // 循环等待回调函数
ROS_INFO("Ready to receive turtle command.");
// 设置循环的频率
ros::Rate loop_rate(10);
    • 根据标志位判断是否发送信息
    if(pubCommand)
{
	geometry_msgs::Twist vel_msg;
	vel_msg.linear.x = 0.5;
	vel_msg.angular.z = 0.2;
	turtle_vel_pub.publish(vel_msg);
}

    标志位的处理在回调函数中进行，一旦标志位为true则进入代码块发送运动信息，标志位为false则不发送信息

    • 回调函数处理
    // service回调函数，输入参数req，输出参数res
bool commandCallback(std_srvs::Trigger::Request  &req,
         			std_srvs::Trigger::Response &res)
{
	pubCommand = !pubCommand;
    // 显示请求数据
    ROS_INFO("Publish turtle velocity command [%s]", pubCommand==true?"Yes":"No");
	// 设置反馈数据
	res.success = true;
	res.message = "Change turtle command state!"
    return true;
}

    pubCommand = !pubCommand;标志位处理

    res.message = "Change turtle command state!"反馈应答数据

    # 三、服务数据定义与使用

    和话题模型类似，在自定义服务数据时也需要自定义一个信息的文件，在话题模型中使用的是.msg文件，但在客户端/服务端模型中使用的是.srv文件

    # 3.1自定义服务数据

    步骤总览：

    • 定义srv文件
    • 在package.xml中添加功能包依赖
    • 在CMakeLists.txt添加编译选项
    • 编译生成语言相关文件

    # 1）定义srv文件

    在功能包下新建文件夹srv，并在文件夹中新建Person.srv文件，写入以下代码

    string name
uint8 age
uint8 sex

uint8 unknown = 0
uint8 male = 1
uint8 female = 2
---
string result

    注意：以---进行分割，该符号以上是Client端发送给Server端的信息，该符号以下是Server端反馈给Client端的信息

    # 2）添加功能包依赖

    在package.xml文件中添加功能包依赖，如下所示：

    <build_depend>message_generation</build_depend>
<exec_depend>message_runtime</exec_depend>

    # 3）添加编译选项

    • 添加message_generation功能包依赖
    find_package(catkin REQUIRED COMPONENTS
  geometry_msgs
  roscpp
  rospy
  std_msgs
  turtlesim

)
    • 在Declare ROS messages, services and actions栏目中添加如下依赖（编译指定头文件）
    add_service_files(FILES Person.srv)
generate_messages(DEPENDENCIES std_msgs)
    • 添加编译依赖

    # 3.2创建Client和Server

    # 1）Client

    /**
 * 该例程将请求/show_person服务，服务数据类型learning_service::Person
 */
#include <ros/ros.h>
#include "learning_service/Person.h"

int main(int argc, char** argv)
{
    // 初始化ROS节点
   ros::init(argc, argv, "person_client");
    // 创建节点句柄
   ros::NodeHandle node;
    // 发现/spawn服务后，创建一个服务客户端，连接名为/spawn的service
   ros::service::waitForService("/show_person");
   ros::ServiceClient person_client = node.serviceClient<learning_service::Person>("/show_person");
    // 初始化learning_service::Person的请求数据
   learning_service::Person srv;
   srv.request.name = "Tom";
   srv.request.age  = 20;
   srv.request.sex  = learning_service::Person::Request::male;
    // 请求服务调用
   ROS_INFO("Call service to show person[name:%s, age:%d, sex:%d]", 
          srv.request.name.c_str(), srv.request.age, srv.request.sex);
   person_client.call(srv);
   // 显示服务调用结果
   ROS_INFO("Show person result : %s", srv.response.result.c_str());
   return 0;
};

    # 2）Server

    /**
 * 该例程将执行/show_person服务，服务数据类型learning_service::Person
 */
#include <ros/ros.h>
#include "learning_service/Person.h"

// service回调函数，输入参数req，输出参数res
bool personCallback(learning_service::Person::Request  &req,
                 learning_service::Person::Response &res)
{
    // 显示请求数据
    ROS_INFO("Person: name:%s  age:%d  sex:%d", req.name.c_str(), req.age, req.sex);
   // 设置反馈数据
   res.result = "OK";
    return true;
}

int main(int argc, char **argv)
{
    // ROS节点初始化
    ros::init(argc, argv, "person_server");
    // 创建节点句柄
    ros::NodeHandle n;
    // 创建一个名为/show_person的server，注册回调函数personCallback
    ros::ServiceServer person_service = n.advertiseService("/show_person", personCallback);
    // 循环等待回调函数
    ROS_INFO("Ready to show person informtion.");
    ros::spin();
    return 0;
}
