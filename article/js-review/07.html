    JavaScript复习-call、apply、bind方法

    Oreo! 2022/7/12 JavaScript

    # 一、call方法

    # 1.1基本概念

    call() 方法使用一个指定的 this 值和单独给出的一个或多个参数来调用一个函数

    第二个参数开始就是传入的参数

    使用实例：

    function say(a, b, c) {
  console.log(this.name)
  console.log(a, b, c)
}

let obj = {name: "myName"}
say.call(obj, 1, 2, 3)
// myName
// 1 2 3

    # 1.2手写实现

    思路：

    • 任何函数都是对Function构造函数的一种实例化 => 相当于在Function.prototype上添加一个新的方法
    • 用call进行绑定相当于在其绑定的对象中添加一个需要绑定的函数，然后再使用obj.func即可

    首先构造一个最基本的框架如下所示：

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newCall = function (obj) {
    console.log(this)
}

say.newCall(object)

    此时Function原型对象上的newCall中的this指向的是function say()，因为是这个function调用的它

    此时需要在调用这个函数的对象上添加该方法，就可以让该方法中的this指向该对象即可实现this的绑定，然后对该绑定的方法进行销毁（由于不能变更对象的原有属性）

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newCall = function (obj) {
    obj.func = this
    obj.func()
    delete obj.func
}

say.newCall(object)

    输出如下所示

    此时完成了一半，因为call还具有传参功能，与普通函数调用不同的是，call函数第一个参数是需要绑定的对象，后面才是该函数需要传入的参数

    • 思路：通过arguments获取后续的参数，将后续参数通过解构赋值的方法传入obj.func()即可
    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newCall = function (obj) {
    obj.func = this
    let newArguments = []
    for (let i = 1; i < arguments.length; i++) {
        newArguments.push(arguments[i])
    }
    obj.func(...newArguments)
    delete obj.func
}

say.newCall(object, 1, 2, 3)

    结果如下所示：

    最后差一步，当传入的对象是null时，新写的call方法将会报错，但是原生的call不会报错，此时在进入函数时添加一个判断：如果传入的是null则指向window即可，最终代码如下：

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newCall = function (obj) {
    obj = obj || window
    obj.func = this
    let newArguments = []
    for (let i = 1; i < arguments.length; i++) {
        newArguments.push(arguments[i])
    }
    obj.func(...newArguments)
    delete obj.func
}

say.newCall(null, 1, 2, 3)

    # 二、apply方法

    # 2.1基本概念

    apply() 方法调用一个具有给定 this 值的函数，以及以一个数组（或一个类数组对象）的形式提供的参数

    注意：该方法的语法和作用与 apply()方法类似，只有一个区别，就是 call() 方法接受的是一个参数列表，而 apply() 方法接受的是一个包含多个参数的数组

    使用实例：

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}
say.apply(object, [1, 2, 3])
// myName
// 1 2 3

    # 2.2手写实现

    • 思路：可以模仿call方法，将后续对arguments的改造改为对数组的传入
    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newApply = function (obj, array) {
    obj = obj || globalThis
    obj.func = this
    if (!array) {
        obj.func()
    } else {
        obj.func(...array)
    }
    delete obj.func
}

say.newApply(object, [1, 2, 3])

    # 三、bind方法

    # 3.1基本概念

    bind() 方法创建一个新的函数，在 bind() 被调用时，这个新函数的 this 被指定为 bind() 的第一个参数，而其余参数将作为新函数的参数，供调用时使用

    该方法同样可以传入多个参数，从第二个开始的参数就是该方法需要传入的参数，当目标函数被调用时，被预置入绑定函数的参数列表中的参数

    function say() {
    console.log(this.name)
}

let object = {name: "myName"}
let newSay = say.bind(object)

say()
// undefined
newSay()
// myName

    # 3.2手写实现

    • 思路：同样的需要在Function.prototype上添加手写的函数，并且该函数的返回值依旧是一个函数，搭建框架如下：
    function say() {
    console.log(this.name)
}

let object = {name: "myName"}

Function.prototype.newBind = function(obj) {
    return function() {}
}

say.newBind(object)()

    和call方法类似，在新写的newBind方法中，由于调用的是say函数，因此newBind方法中的this指向调用该函数的方法say()

    但是由于函数里面返回的也是一个函数，在执行中容易导致this的丢失，需要将其保存

    function say() {
    console.log(this.name)
}

let object = {name: "myName"}

Function.prototype.newBind = function(obj) {
    let that = this
    return function() {}
}

say.newBind(object)()
    • 思路：返回的方法主要的功能就是将that中保存的方法指向传入的对象即可
    function say() {
    console.log(this.name)
}

let object = {name: "myName"}

Function.prototype.newBind = function(obj) {
    let that = this
    return function() {
        that.apply(obj)
    }
}

say.newBind(object)()

    处理传入的函数参数，可以巧妙运用call方法，将Array原型上的slice方法绑定到arguments对象中实行类数组的切割并赋给return出来的函数

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newBind = function(obj) {
    let that = this
    // 巧妙运用call，将Array的slice方法传给arguments对象进行使用
    let arr = Array.prototype.slice.call(arguments, 1)
    return function() {
        that.apply(obj, arr)
    }
}

say.newBind(object, 1, 2, 3)()

    由于bind函数是具有柯里化特性的，也就是说以上面为例say.newBind(object, 1, 2, 3)()的第二个括号里是可以传入参数的，因此可以将return内函数的arguments与外部的arguments，打印效果如下

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newBind = function(obj) {
    // 由于函数里面返回的也是一个函数，在执行中容易导致this的丢失，需要将其保存
    let that = this
    // 巧妙运用call，将Array的slice方法传给arguments对象进行使用
    let arr1 = Array.prototype.slice.call(arguments, 1)
    console.log(arr1)
    return function() {
        let arr2 = Array.prototype.slice.call(arguments)
        console.log(arr2)
        that.apply(obj)
    }
}

say.newBind(object, 1, 2, 3)(4, 5, 6)

    将两个数组合并并传入apply方法中

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newBind = function(obj) {
    // 由于函数里面返回的也是一个函数，在执行中容易导致this的丢失，需要将其保存
    let that = this
    // 巧妙运用call，将Array的slice方法传给arguments对象进行使用
    let arr1 = Array.prototype.slice.call(arguments, 1)
    return function() {
        let arr2 = Array.prototype.slice.call(arguments)
        let arr = arr1.concat(arr2)
        that.apply(obj, arr)
    }
}

say.newBind(object, 1, 2)(3)

    此时基本功能已经完成，但是bind方法可以配合new使用，但是this会失效。要判断有没有使用new使用到了原型对象的概念，由于有了名字的函数才能进行关系的确认，此处将其改造为具名函数

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newBind = function(obj) {
    // 由于函数里面返回的也是一个函数，在执行中容易导致this的丢失，需要将其保存
    let that = this
    // 巧妙运用call，将Array的slice方法传给arguments对象进行使用
    let arr1 = Array.prototype.slice.call(arguments, 1)
    let newFunction = function() {
        let arr2 = Array.prototype.slice.call(arguments)
        let arr = arr1.concat(arr2)
        that.apply(obj, arr)
    }
    return newFunction
}

say.newBind(object, 1, 2)(3)

    但是此时通过new创造一个新实例时，newFunction函数中的this将会是newFunction的实例，如下所示

    也就是说此时this不是say函数，而是一个newFunction方法的实例，如果是现在这种情况，就不需要保存前面的that变量，处理如下所示：

    function say(a, b, c) {
    console.log(this.name)
    console.log(a, b, c)
}

let object = {name: "myName"}

Function.prototype.newBind = function(obj) {
    // 由于函数里面返回的也是一个函数，在执行中容易导致this的丢失，需要将其保存
    let that = this
    // 巧妙运用call，将Array的slice方法传给arguments对象进行使用
    let arr1 = Array.prototype.slice.call(arguments, 1)
    let newFunction = function() {
        let arr2 = Array.prototype.slice.call(arguments)
        let arr = arr1.concat(arr2)
        // 使用this判断有没有使用new构造符
        if (this instanceof newFunction) {
            // 有的话就将apply的this绑定到新实例上
            that.apply(this, arr)
        } else {
            that.apply(obj, arr)
        }
    }
    // 利用原型链将两者连接起来：将say原型对象复制给new出来的原型对象
    newFunction.prototype = that.prototype
    return newFunction
}

say.newBind(object, 1, 2)(3)
